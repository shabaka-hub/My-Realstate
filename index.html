    <script>

        const apiKey = ""; // Set by environment


        async function callGemini(prompt) {

            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;

            const payload = {

                contents: [{ parts: [{ text: prompt }] }],

                systemInstruction: { parts: [{ text: "You are a luxury real estate copywriter. Write a compelling, elegant, and high-end property description for a real estate listing in Windsor, Ontario. Use cinematic language. Keep it between 100-150 words." }] }

            };


            for (let i = 0; i < 5; i++) {

                try {

                    const response = await fetch(url, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });

                    const data = await response.json();

                    return data.candidates?.[0]?.content?.parts?.[0]?.text;

                } catch (err) {

                    await new Promise(r => setTimeout(r, Math.pow(2, i) * 1000));

                }

            }

        }


        async function generateDescription() {

            const details = document.getElementById('propDetails').value;

            const vibe = document.getElementById('propVibe').value;

            if (!details) return;


            const btn = document.getElementById('generateBtn');

            btn.disabled = true;

            btn.innerHTML = `<span class="spinner"></span> Thinking...`;


            try {

                const result = await callGemini(`Property: ${details}. Style: ${vibe}.`);

                document.getElementById('aiPlaceholder').classList.add('hidden');

                document.getElementById('aiContent').classList.remove('hidden');

                document.getElementById('aiOutput').innerText = result;

            } catch (e) {

                alert("AI error. Try again.");

            } finally {

                btn.disabled = false;

                btn.innerHTML = `✨ Generate Description`;

            }

        }


        function copyDescription() {

            const text = document.getElementById('aiOutput').innerText;

            const el = document.createElement('textarea');

            el.value = text;

            document.body.appendChild(el);

            el.select();

            document.execCommand('copy');

            document.body.removeChild(el);

            alert("Copied!");

        }
